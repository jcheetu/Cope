DROP Database IF EXISTS COPE;
create Database COPE;
Use COPE;

DROP TABLE IF EXISTS Client;
CREATE TABLE Client(
   ClientId BIGINT AUTO_INCREMENT PRIMARY KEY,
   MarketName VARCHAR(40),
   EmailId VARCHAR(255),
   Phone VARCHAR(255),
   BussinessCategory VARCHAR(255),
   Country VARCHAR(255),
   City VARCHAR(255),
   OnboardingDate DATE,
   CreatedOn DATE,
   UpdatedOn DATE
);

DROP TABLE IF EXISTS Plan;
CREATE TABLE Plan(
   PlanId BIGINT AUTO_INCREMENT PRIMARY KEY,
   PlanName VARCHAR(40),
   Connects INT,
   Price INT,
   Currency CHAR(5),
   PlanLevel VARCHAR(255),
   CreatedOn DATE,
   UpdatedOn DATE
);



DROP TABLE IF EXISTS ClientPlan;
CREATE TABLE ClientPlan(
   Id BIGINT AUTO_INCREMENT PRIMARY KEY,
   ClientId BIGINT,
   PlanId INT,
   LevelState CHAR(2), 
   Currency CHAR(5),
   PlanLevel VARCHAR(255),
   CreatedOn DATE,
   UpdatedOn DATE
);

DROP TABLE IF EXISTS AI;
CREATE TABLE AI(
   AIID BIGINT AUTO_INCREMENT PRIMARY KEY,
   AIName varchar(255),
   AI CHAR(5)
  );
DROP TABLE IF EXISTS ClientToken;
CREATE TABLE ClientToken(
   TokenID BIGINT AUTO_INCREMENT PRIMARY KEY,
   ClientID BIGINT,
   AI CHAR(5),
   accessToken VARCHAR(255),
   COPEToken VARCHAR(255),
   UserName VARCHAR(255),
   Password VARCHAR(255),
   CreatedOn DATE,
   UpdatedOn DATE
);

DROP TABLE IF EXISTS Currency

CREATE TABLE Currency(
   CurrencyID BIGINT AUTO_INCREMENT PRIMARY KEY,
   Currency Varchar(255),
   CurrencyCode CHAR(5)
);

DROP TABLE IF EXISTS Logger
CREATE TABLE Logger(
   LogID BIGINT AUTO_INCREMENT PRIMARY KEY,
   Request Varchar(255),
   LogType CHAR(5)
);
